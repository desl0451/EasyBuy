jQuery滑动条插件 soSlider 1.0 (点击下载)

所有参数：

$(obj).soSlider({
    type : 1, //类型:1||2，1个滑动点，2个滑动点，默认为1个滑动点
    bit :1, //进位步长，默认为1
    data:null, //'min,max' 字符串格式，用逗号隔开数值，最小数和最大数
    defaultPos:null, //'point1,point2' 字符串格式，默认数值，1个滑动点1个数值
    showRange : true, //显示滑动区间
    initEvent : function (){}, //加载执行事件， initEvent : function(v1,v2){}提供2点对应数值
    mMoveEvent : function () {}, //移动执行事件， mMoveEvent : function(v1,v2){}提供2点对应数值
    mUpEvent : function (){} //鼠标按起执行事件， mUpEvent : function(v1,v2){}提供2点对应数值
});

1- 单点模式 type : 1

    $('#slider1').soSlider({
        data :'0,500', //初始数据范围
        bit:50, //以50为进制
        initEvent:function (v) { //初始事件
            $('#s-sliderNum0').html(v);
        },
        mMoveEvent:function (v) { //移动事件
            $('#s-sliderNum0').html(v);
        }
    });

注：默认为单点模式，不设置默认位置点，滑动点定位在起点和终点

2- 双点范围模式 type : 2

    $('#slider2').soSlider({
        type:2, //双点范围模式
        data :'800,5000', //初始数据范围
        defaultPos : '1000,4000', //默认双点数据
        bit:20, //以20为进制
        initEvent : function (d1,d2) { //初始事件
            $('#s-sliderNum').html(d1);
            $('#s-sliderNum2').html(d2);
        },
        mMoveEvent : function (v1,v2) { //移动事件
            $('#s-sliderNum').html(v1);
            $('#s-sliderNum2').html(v2);
        },
        mUpEvent : function (v1,v2) { //按起事件
            $('#s-sliderNum3').html(v1);
            $('#s-sliderNum4').html(v2);
        }
    });
